68 class MemoryManager : DeletableObjectBaseClass
69 {
70  //Keeps track of all objects created using the ``New'' method
71  //When the MemoryManager is deleted, everything it manages is deleted too
72 public:
73  const static unsigned int BlockSize = 62;
74  struct MemoryBlock
75  {
76  DeletableObjectBaseClass* Pointers[BlockSize];
77  size_t count = 0;
78  MemoryBlock* next;
79  MemoryBlock() : next(0) {}
80  MemoryBlock(MemoryBlock* next) : next(next) {}
81  };
82  MemoryBlock* first;
83  MemoryManager() : first(new MemoryBlock()) {}
84  template<typename T, typename... In> T* New(In... inputs)
85  {
86  T* pointer = new T(inputs...);
87  unsigned long insert_location = first->count;
88  if(insert_location < BlockSize){
89  first->count = insert_location + 1;
90  first->Pointers[insert_location] = pointer;
91  }
92  else {
93  first = new MemoryBlock(first);
94  first->count = 1;
95  first->Pointers[0] = pointer;
96  }
97  return pointer;
98  }
99  ~MemoryManager()
100  { //Deletes everything allocated with MemoryManager::New
101  MemoryBlock* current = first;
102  MemoryBlock* next = current->next;
103  DeletableObjectBaseClass** scan = current->Pointers;
104  unsigned int count = current->count;
105  for(unsigned int i = 0; i < count; ++i) { delete *scan++; }
106  delete current;
107  current = next;
108  while(current){
109  scan = current->Pointers;
110  next = current->next;
111  for(unsigned int i = 0; i < BlockSize; ++i){
112  delete *scan++;
113  }
114  delete current;
115  current = next;
116  }
117  }
118 };

316  Evaluatable<bool>* condition;
317  Evaluatable<Out>* IfTrue;
318  Evaluatable<Out>* IfFalse;
319  operator Out() const { return (*condition) ? (*IfTrue) : (*IfFalse); }
320  Conditional(Evaluatable<bool>* condition, Evaluatable<Out>* WhenTrue, Evaluatable<Out>* WhenFalse)
321  : condition(condition), IfTrue(WhenTrue), IfFalse(WhenFalse) {}
322  Evaluatable<Out>* optimize(MemoryManager& m){
323  Evaluatable<bool>* condition = this->condition->optimize(m);
324  Evaluatable<Out>* IfTrue = this->IfTrue->optimize(m);
325  Evaluatable<Out>* IfFalse = this->IfFalse->optimize(m);
326  if(auto ConstCondition = dynamic_cast<GenericConstant<bool>*>(condition)){
327  if(ConstCondition->value) { return IfTrue; }
328  else return IfFalse;
329  }
330  this->condition = condition;
331  this->IfTrue = IfTrue;
332  this->IfFalse = IfFalse;
333  return this;

369  int Next(char open, char close) {
370  const char* f = this->first;
371  if (f[pos] == open) {
372  int depth = 1;
373  while (depth != 0 && pos < length) {
374  char c = f[++pos];
375  if (c == open) depth++;
376  else if (c == close) depth--;
377  }
378  return depth;
379  } else pos++;
380  return 0;
381  }
382  int Prev(char open, char close) {
383  const char* f = this->first;
384  if (f[pos] == close) {
385  int depth = 1;
386  while (depth != 0 && pos > 0) {
387  char c = f[--pos];
388  if (c == close) depth++;
389  else if (c == open) depth--;
390  }
391  return depth;
392  } else pos--;
393  return 0;
394  }

480  RealValuedExpression* ParseFunction(Reader r) {
481  char c = *r.first;
482  if(IsDigit(c) || c == '(' || c == '-' || r.length < 3) return 0;
483  while(c = r.first[r.pos], c != '(' && c != ' ' && r.pos < r.length) {
484  if(c != '*' && c != '/' && c != '^' && c != '+' && c != '=' && c != '<' && c != '~' && c != '>')
485  r.pos++;
486  else return 0;
487  }
488  if(r >> '('){
489  int p = r.pos;
490  int depth = r.Next('(', ')');
491  if(depth > 0) throw std::invalid_argument("Mismatched parethesis in " + r.MakeString());
492  if(r.pos < r.length - 1) return 0;
493  Reader func_name = r.sub(r.first, p);
494  if(IsCMathFunc(func_name)){
495  RealValuedExpression* exp = ParseEquation(r.sub(p, r.length));
496  return GetCMathFunc(func_name, exp, mem);
497  }
498  }
499  return 0;
500  }
501  RealValuedExpression* ParseOperation(Reader r){
502  if(r >> '?') {
503  int p1 = r.pos;
504  int depth = r.Next('?', ':');
505  if(depth > 0) throw std::invalid_argument("Couldn't find matching : for ternary operator in " + r.MakeString());
506  if(r.pos != r.length) {
507  return mem.New<RealValuedConditional>(ParseBooleanExpression(r.sub(r.first, p1)),
508  ParseEquation(r.sub(p1 + 1, r.pos)),
509  ParseEquation(r.sub(r.pos + 1, r.length)));
510  }
511  }
512  if(r << '%') return mem.New<DoubleMod>(
513  ParseEquation(r.sub(r.first, r.pos)),
514  ParseEquation(r.sub(r.pos + 1, r.length)));
515 
516  if(r << '+') return mem.New<DoubleAdd>(
517  ParseEquation(r.sub(r.first, r.pos)),
518  ParseEquation(r.sub(r.pos + 1, r.length)));
519 
520  if(r << '-') {
521  //We need to ensure that we found a minus sign and not a value negation
522  while(r.pos > 0 && (r.first[r.pos] == '-' || r.first[r.pos] == ' ')) r.pos--;
523  if(r.pos > 0 || r.first[r.pos] != '-') {
524  char c = r.first[r.pos];
525  if(c != '*' && c != '/' && c != '^' && c != 'e' && c != '=' && c != '<' && c != '~' && c != '>')
526  {
527  while(r.first[r.pos] != '-') r.pos++;//Moves it back to the last minus sign
528  return mem.New<DoubleSubtract>(
529  ParseEquation(r.sub(r.first, r.pos)),
530  ParseEquation(r.sub(r.pos + 1, r.length)));
531  }
532  }
533  }
534 
535  if(r << '*') return mem.New<DoubleMultiply>(
536  ParseEquation(r.sub(r.first, r.pos)),
537  ParseEquation(r.sub(r.pos + 1, r.length)));
538 
539  if(r << '/') return mem.New<DoubleDivide>(
540  ParseEquation(r.sub(r.first, r.pos)),
541  ParseEquation(r.sub(r.pos + 1, r.length)));
542 
543  if(r >> '^') {
544  RealValuedExpression* base = ParseEquation(r.sub(r.first, r.pos));
545  Reader power_text = r.sub(r.pos + 1, r.length);
546  ApplyNecessaryFormatting(power_text);
547  if(IsInteger(power_text))
548  return mem.New<DoubleToIntegerPow>(base, mem.New<GenericConstant<int>>(strtol(power_text.first, NULL, 10)));
549  else return mem.New<DoublePow>(base, ParseEquation(power_text));
550  }
551 
552  if(*r.first == '-') {
553  return mem.New<GenericNeg<double, double>>(ParseEquation(r.sub(1, r.length)));
554  }
555 
556  return 0;
557  }
