101  /* README
102  All processors calling MPI_Comm_split with the same comm_color will
103  be grouped into the same communicator group. This means that this ReductionClique_t
104  will share a clique_comm with all the other ReductionClique_ts created with the
105  given comm_color. The way this works is that if a set of ranks R all share
106  a set of nodes N (so that every rank in R has every node in N), all ranks
107  in R will be grouped into a single MPI_Comm (identified by the comm_color).
108  The bookkeeping for this new MPI_Comm is stored in a ReductionClique_t struct,
109  along with the bookkeeping necessary to call Allreduce to reduce the data
110  for all the nodes in N.
111  See http://mpitutorial.com/tutorials/introduction-to-groups-and-communicators/
112  for a reference
113  */

217  // Returns false if the currently active asynchronous reduce request hasn't completed

234  // completed Returns false if the currently active asynchronous reduce request hasn't completed

272  cliques.reserve(RawCliqueData.size());
273  {
274  auto RawCliqueDataIterator = RawCliqueData.begin();
275  for (int color = 0; color < max_color; ++color)
276  {

278  MPI_Comm_split(context.get_comm(), MPI_UNDEFINED, 0, &discard_comm);
279  else if ((*RawCliqueDataIterator).first == color)

281  auto& color_nodelst_pair = *RawCliqueDataIterator;
282  ++RawCliqueDataIterator;

284  int color = color_nodelst_pair.first;
285 
286  int buffersize = maximum_field_size * nodelist.size();
287  cliques.emplace_back(nodelist, buffersize, color, context);
288  }
289  else
290  throw std::invalid_argument("Something went horribly wrong in ReductionInfo");

297  template<int field_size>
298  void Reduce(double* data)
299  {
300  int n_cliques = cliques.size();
301  int n_unfinished = n_cliques;
302  for (int i = 0; i < n_cliques; ++i)
303  {
304  cliques[i].asyncreduce_initialize<field_size>(data);
305  unfinished[i] = i;
306  }
307  while (n_unfinished > 0)
308  {
309  for (int i = 0; i < n_unfinished; ++i)
310  {
311  int clique_id = unfinished[i];
312  // If the clique finalizes successfully, remove it from the list of cliques to finalize.
313  if (cliques[clique_id].asyncreduce_finalize<field_size>(data))
314  {
315  unfinished[i] = unfinished[n_unfinished - 1];
316  n_unfinished--;
317  i--;
318  }
319  }
320  }
321  }

366 pair<color_t, vector<pair<color_t, globalid_v>>> DistributeCliqueInformation(
367  const vector<pair<rank_v, globalid_v>>& clique_info,
368  const mpicontext& context)
369 {
370  pair<color_t, vector<pair<color_t, globalid_v>>> output;
371  vector<pair<color_t, globalid_v>>& my_clique_info = output.second;
372  if (context.get_rank() == 0)
373  {
374  vector<color_v> clique_assignments(context.get_size());
375  int n_cliques = clique_info.size();
376  for (color_t color = 0; color < n_cliques; ++color)
377  {
378  const rank_v& ranks_in_comm = clique_info[color].first;
379  for (rankid r : ranks_in_comm)
380  {
381  clique_assignments[r].push_back(color);
382  }
383  }
384  // puts("Calculated clique_assignments");
385  vector<int> send_buffer;
386  pair<color_t, vector<pair<color_t, const globalid_v*>>> thing_to_send;
387  thing_to_send.first = clique_info.size();
388  vector<pair<color_t, const globalid_v*>>& rank_clique_info = thing_to_send.second;
389  for (rankid r = 1; r < context.get_size(); ++r)
390  {
391  color_v& assignment = clique_assignments[r];
392  rank_clique_info.clear();
393  rank_clique_info.reserve(assignment.size());
394  for (color_t color : assignment)
395  rank_clique_info.emplace_back(color, &(clique_info[color].second));
396 
397  serialization::pack(thing_to_send, send_buffer);
398  // printf("DistributeCliqueInformation: Sending %d ints to rank %d\n", send_buffer.size(), r);
399  MPI_Send(send_buffer.data(),
400  send_buffer.size(),
401  MPI_INT,
402  r,
403  DistributeCliqueInformationTag,
404  context.get_comm());
405  // printf("DistributeCliqueInformation: Sent to rank %d\n", r);
406  }
407  {
408  output.first = clique_info.size();
409  color_v& assignment = clique_assignments[0];
410  my_clique_info.reserve(assignment.size());
411  for (color_t color : assignment)
412  my_clique_info.emplace_back(color, clique_info[color].second);
413  }
414  }
415  else
416  {
417  vector<int> recv_buffer;
418  MPI_Status recv_status;
419  int recv_size;
420  // printf("DistributeCliqueInformation rank %d: probing size\n", context.get_rank());
421  MPI_Probe(0, DistributeCliqueInformationTag, context.get_comm(), &recv_status);
422  MPI_Get_count(&recv_status, MPI_INT, &recv_size);
423  // printf("DistributeCliqueInformation rank %d: preparing to recieve %d ints\n",
424  // context.get_rank(), recv_size);
425  recv_buffer.resize(recv_size);
426  MPI_Recv(recv_buffer.data(),
427  recv_size,
428  MPI_INT,
429  0,
430  DistributeCliqueInformationTag,
431  context.get_comm(),
432  MPI_STATUS_IGNORE);
433  serialization::unpack(output, recv_buffer);
434  }
435  // printf("rank %d clique info: ", context.get_rank());
436  // std::cout << output << std::endl;
437  // printf("DistributeCliqueInformation rank %d: returning my_clique_info\n", context.get_rank());
438  return output;
439 }
440 pair<color_t, vector<pair<color_t, globalid_v>>> FindMyCliques(
441  const std::vector<int>& global_node_ids,
442  std::unique_ptr<int[]>& node_counts,
443  const mpicontext& context)

448  typedef std::vector<int> nodelst;

464  for (int i = 0; i < max_node_id; ++i)
465  {
466  if (ranks_per_node[i] > 1)
467  ranks_sharing_each_node[i].first.reserve(ranks_per_node[i]);
468  ranks_sharing_each_node[i].second = i;
469  }
470  for (int i : global_node_ids)
471  if (ranks_per_node[i] > 1)
472  ranks_sharing_each_node[i].first.push_back(0);
473 
474  // puts("Allocated ranks_sharing_each_node");

476  for (int rank = 1; rank < context.get_size(); ++rank)

487  {
488  int node_id = gni_other[i];

518  // MPI_Barrier(context.get_comm());
519  }

521  vector<pair<rank_v, globalid_v>> all_cliques_empty = {};
