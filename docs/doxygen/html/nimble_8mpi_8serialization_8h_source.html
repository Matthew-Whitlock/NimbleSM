258 size_t pack(const T& object, std::vector<dataT>& vect)
259 {
260  static_assert(std::is_pod<dataT>::value, "vector must be of POD type");
261  size_t size = serial_t<T>::size(object);
262  vect.resize(size / sizeof(dataT) + 1);

264  serial_t<T>::pack(object, dataptr);
