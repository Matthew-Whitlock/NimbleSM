67  const char* str2) {
68  int len1 = StringLength(str1);
69  int len2 = StringLength(str2);
70  int len = len1 < len2 ? len1 : len2;
71  bool are_equal = true;

73  if (str1[i] != str2[i]) {

119  NIMBLE_INLINE_FUNCTION
120  void AddParameter(const char* parameter_name, double parameter_value) {
121  int len = StringLength(parameter_name);
122  for (int i=0 ; i<len ; ++i) {
