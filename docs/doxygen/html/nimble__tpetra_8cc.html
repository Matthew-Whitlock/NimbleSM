787  + (F[K_F_YY] - identity[K_F_YY]) * (reference_coordinate[i_y] - rve_center.at(1))

794  }
795  }

797  // check convergence
798  for (int i=0 ; i<num_unknowns ; ++i) {
799  internal_force[i] = 0.0;
800  }

821  elem_data_np1,
822  data_manager,

824  }
825  for (int i=0 ; i<linear_system_num_nodes*dim ; i++) {
826  residual_vector[i] = 0.0;
827  }
828  for (int n=0 ; n<num_nodes ; n++) {
829  int ls_id = linear_system_global_node_ids[n];
830  for (int dof=0 ; dof<dim ; dof++) {
831  int local_index = n * dim + dof;
832  int ls_index = ls_id * dim + dof;
833  if (ls_index < 0 || ls_index > residual_vector.size() - 1) {
834  throw std::logic_error("\nError: Invalid index into residual vector in QuasistaticTimeIntegrator().\n");
835  }
836  residual_vector[ls_index] += -1.0 * internal_force[local_index];
837  }
838  }
839  bc.ModifyRHSForKinematicBC(linear_system_global_node_ids.data(), residual_vector.data());

841  for (int i=0 ; i<num_nodes ; i++) {
842  double f = std::fabs(residual_vector[i]);
843  l2_norm += f * f;
844  if (f > infinity_norm) {
845  infinity_norm = f;
846  }
847  }
848  l2_norm = std::sqrt(l2_norm);
849  residual = l2_norm + 20.0 * infinity_norm;
850

852
