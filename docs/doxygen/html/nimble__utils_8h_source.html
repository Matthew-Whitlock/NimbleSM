413 template <typename ScalarT>
414 NIMBLE_INLINE_FUNCTION
415 ScalarT Invert_Full33(const ScalarT* mat, ScalarT* inv) {
416 
417  ScalarT minor0 = mat[K_F_YY] * mat[K_F_ZZ] - mat[K_F_YZ] * mat[K_F_ZY];
418  ScalarT minor1 = mat[K_F_YX] * mat[K_F_ZZ] - mat[K_F_YZ] * mat[K_F_ZX];
419  ScalarT minor2 = mat[K_F_YX] * mat[K_F_ZY] - mat[K_F_YY] * mat[K_F_ZX];
420  ScalarT minor3 = mat[K_F_XY] * mat[K_F_ZZ] - mat[K_F_XZ] * mat[K_F_ZY];
421  ScalarT minor4 = mat[K_F_XX] * mat[K_F_ZZ] - mat[K_F_ZX] * mat[K_F_XZ];
422  ScalarT minor5 = mat[K_F_XX] * mat[K_F_ZY] - mat[K_F_XY] * mat[K_F_ZX];
423  ScalarT minor6 = mat[K_F_XY] * mat[K_F_YZ] - mat[K_F_XZ] * mat[K_F_YY];
424  ScalarT minor7 = mat[K_F_XX] * mat[K_F_YZ] - mat[K_F_XZ] * mat[K_F_YX];
425  ScalarT minor8 = mat[K_F_XX] * mat[K_F_YY] - mat[K_F_XY] * mat[K_F_YX];
426  ScalarT det = mat[K_F_XX] * minor0 - mat[K_F_XY] * minor1 + mat[K_F_XZ] * minor2;
427 
428  if (det < 0.0) {
429  //throw std::logic_error("Error in Invert_Full33(), singular matrix");
430  }
431 
432  inv[K_F_XX] = minor0 / det;
433  inv[K_F_XY] = -1.0 * minor3 / det;
434  inv[K_F_XZ] = minor6 / det;
435  inv[K_F_YX] = -1.0 * minor1 / det;
436  inv[K_F_YY] = minor4 / det;
437  inv[K_F_YZ] = -1.0 * minor7 / det;
438  inv[K_F_ZX] = minor2 / det;
439  inv[K_F_ZY] = -1.0 * minor5 / det;
440  inv[K_F_ZZ] = minor8 / det;
441 
442  return det;
443 }

445 //! R^N logarithmic map using Baker-Campbell-Hausdorff (BCH) expansion (4 terms)
446 // All arguments are assumed to be full tensors (9 components)
447 template <typename ScalarT>
448 NIMBLE_INLINE_FUNCTION
449 void BCH(const ScalarT* const x,
450  const ScalarT* const y,
451  ScalarT* const result) {
452 
453  // adapted from MiniTensor_LinearAlgebra.t.h bch()
454 
455  ScalarT temp1[9], temp2[9], temp3[9];
456 
457  // first order term
458  // x + y
459  Sum_Full33_Full33(x, y, result);
460 
461  // second order term
462  // 0.5*(x*y - y*x)
463  Mult_Scalar_Full33_Full33(0.5, x, y, temp1);
464  Mult_Scalar_Full33_Full33(-0.5, y, x, temp2);
465  Sum_Full33_Full33(result, temp1, result);
466  Sum_Full33_Full33(result, temp2, result);
467 
468  // third order term
469  // 1.0/12.0 * (x*x*y - 2.0*x*y*x + x*y*y + y*x*x - 2.0*y*x*y + y*y*x)
470  // 1/12 * x*x*y
471  Mult_Full33_Full33(x, y, temp1);
472  Mult_Scalar_Full33_Full33(1.0/12.0, x, temp1, temp2);
473  Sum_Full33_Full33(result, temp2, result);
474  // -1/6 * x*y*x
475  Mult_Full33_Full33(y, x, temp1);
476  Mult_Scalar_Full33_Full33(-1.0/6.0, x, temp1, temp2);
477  Sum_Full33_Full33(result, temp2, result);
478  // 1/12 * x*y*y
479  Mult_Full33_Full33(y, y, temp1);
480  Mult_Scalar_Full33_Full33(1.0/12.0, x, temp1, temp2);
481  Sum_Full33_Full33(result, temp2, result);
482  // 1/12 * y*x*x
483  Mult_Full33_Full33(x, x, temp1);
484  Mult_Scalar_Full33_Full33(1.0/12.0, y, temp1, temp2);
485  Sum_Full33_Full33(result, temp2, result);
486  // -1/6 * y*x*y
487  Mult_Full33_Full33(x, y, temp1);
488  Mult_Scalar_Full33_Full33(-1.0/6.0, y, temp1, temp2);
489  Sum_Full33_Full33(result, temp2, result);
490  // 1/12 * y*y*x
491  Mult_Full33_Full33(y, x, temp1);
492  Mult_Scalar_Full33_Full33(1.0/12.0, y, temp1, temp2);
493  Sum_Full33_Full33(result, temp2, result);
494 
495  // fourth order term
496  // 1.0/24.0 * (x*x*y*y - 2.0*x*y*x*y + 2.0*y*x*y*x - y*y*x*x);
497  // 1.0/24.0 * x*x*y*y
498  Mult_Full33_Full33(y, y, temp1);
499  Mult_Full33_Full33(x, temp1, temp2);
500  Mult_Scalar_Full33_Full33(1.0/24.0, x, temp2, temp3);
501  Sum_Full33_Full33(result, temp3, result);
502  // -2.0/24.0 * x*y*x*y
503  Mult_Full33_Full33(x, y, temp1);
504  Mult_Full33_Full33(y, temp1, temp2);
505  Mult_Scalar_Full33_Full33(-2.0/24.0, x, temp2, temp3);
506  Sum_Full33_Full33(result, temp3, result);
507  // 2.0/24.0 * y*x*y*x
508  Mult_Full33_Full33(y, x, temp1);
509  Mult_Full33_Full33(x, temp1, temp2);
510  Mult_Scalar_Full33_Full33(2.0/24.0, y, temp2, temp3);
511  Sum_Full33_Full33(result, temp3, result);
512  // -1.0/24.0 * y*y*x*x
513  Mult_Full33_Full33(x, x, temp1);
514  Mult_Full33_Full33(y, temp1, temp2);
515  Mult_Scalar_Full33_Full33(-1.0/24.0, y, temp2, temp3);
516  Sum_Full33_Full33(result, temp3, result);
517 }

565 template <typename ScalarT>
566 NIMBLE_INLINE_FUNCTION
567 void Eigen_Sym33_NonUnit(const ScalarT* const tensor,
568  ScalarT& eval0, ScalarT& eval1, ScalarT& eval2,
569  ScalarT* const evec0, ScalarT* const evec1, ScalarT* const evec2) {
570 
571  ScalarT cxx = tensor[K_S_XX];
572  ScalarT cyy = tensor[K_S_YY];
573  ScalarT czz = tensor[K_S_ZZ];
574  const ScalarT& cxy = tensor[K_S_XY];
575  const ScalarT& cyz = tensor[K_S_YZ];
576  const ScalarT& czx = tensor[K_S_ZX];
577 
578  const ScalarT c1 = (cxx + cyy + czz)/ScalarT(3.0);
579 
580  cxx -= c1;
581  cyy -= c1;
582  czz -= c1;
583 
584  const ScalarT cxy_cxy = cxy*cxy;
585  const ScalarT cyz_cyz = cyz*cyz;
586  const ScalarT czx_czx = czx*czx;
587  const ScalarT cxx_cyy = cxx*cyy;
588 
589  const ScalarT c2 = cxx_cyy + cyy*czz + czz*cxx - cxy_cxy - cyz_cyz - czx_czx;
590 
591  const ScalarT ThreeOverA = ScalarT(-3.0)/c2;
592  const ScalarT sqrtThreeOverA = std::sqrt(ThreeOverA);
593 
594  const ScalarT c3 = cxx*cyz_cyz + cyy*czx_czx - ScalarT(2.0)*cxy*cyz*czx + czz*(cxy_cxy - cxx_cyy);
595 
596  const ScalarT rr = ScalarT(-0.5)*c3*ThreeOverA*sqrtThreeOverA;
597 
598  const ScalarT arg = Minimum(std::abs(rr),ScalarT(1.0));
599 
600  const ScalarT cos_thd3 = Cos_Of_Acos_Divided_By_3(arg);
601 
602  const ScalarT two_cos_thd3 = ScalarT(2.0)*MultiplySign(cos_thd3,rr);
603 
604  eval2 = two_cos_thd3/sqrtThreeOverA;
605 
606  ScalarT crow0[3] = {cxx - eval2, cxy, czx };
607  ScalarT crow1[3] = {cxy, cyy - eval2, cyz };
608  ScalarT crow2[3] = {czx, cyz, czz - eval2};
609 
610  //
611  // do QR decomposition with column pivoting
612  //
613  const ScalarT k0 = crow0[0]*crow0[0] + cxy_cxy + czx_czx;
614  const ScalarT k1 = cxy_cxy + crow1[1]*crow1[1] + cyz_cyz;
615  const ScalarT k2 = czx_czx + cyz_cyz + crow2[2]*crow2[2];
616 
617  ScalarT k_row1[3];
618  ScalarT row2[3];
619  ScalarT row3[3];
620 
621  // returns zero or nan
622  const bool k0gk1 = k1<=k0;
623  const bool k0gk2 = k2<=k0;
624  const bool k1gk2 = k2<=k1;
625 
626  const bool k0_largest = k0gk1 && k0gk2;
627  const bool k1_largest = k1gk2 && !k0gk1;
628  const bool k2_largest = !(k0_largest || k1_largest);
629 
630  k_row1[0] = if_then_else_zero(k0_largest, crow0[0])
631  + if_then_else_zero(k1_largest, crow1[0])
632  + if_then_else_zero(k2_largest, crow2[0]);
633 
634  k_row1[1] = if_then_else_zero(k0_largest, crow0[1])
635  + if_then_else_zero(k1_largest, crow1[1])
636  + if_then_else_zero(k2_largest, crow2[1]);
637 
638  k_row1[2] = if_then_else_zero(k0_largest, crow0[2])
639  + if_then_else_zero(k1_largest, crow1[2])
640  + if_then_else_zero(k2_largest, crow2[2]);
641 
642  row2[0] = if_then_else(k0_largest, crow1[0], crow0[0]);
643  row2[1] = if_then_else(k0_largest, crow1[1], crow0[1]);
644  row2[2] = if_then_else(k0_largest, crow1[2], crow0[2]);
645 
646  row3[0] = if_then_else(k2_largest, crow1[0], crow2[0]);
647  row3[1] = if_then_else(k2_largest, crow1[1], crow2[1]);
648  row3[2] = if_then_else(k2_largest, crow1[2], crow2[2]);
649 
650  const ScalarT ki_ki = ScalarT(1.0) / ( if_then_else_zero(k0_largest, k0)
651  + if_then_else_zero(k1_largest, k1)
652  + if_then_else_zero(k2_largest, k2) );
653 
654  const ScalarT ki_dpr1 = ki_ki*(k_row1[0]*row2[0] + k_row1[1]*row2[1] + k_row1[2]*row2[2]);
655  const ScalarT ki_dpr2 = ki_ki*(k_row1[0]*row3[0] + k_row1[1]*row3[1] + k_row1[2]*row3[2]);
656 
657  row2[0] -= ki_dpr1*k_row1[0];
658  row2[1] -= ki_dpr1*k_row1[1];
659  row2[2] -= ki_dpr1*k_row1[2];
660 
661  row3[0] -= ki_dpr2*k_row1[0];
662  row3[1] -= ki_dpr2*k_row1[1];
663  row3[2] -= ki_dpr2*k_row1[2];
664 
665  const ScalarT a0 = row2[0]*row2[0] + row2[1]*row2[1] + row2[2]*row2[2];
666  const ScalarT a1 = row3[0]*row3[0] + row3[1]*row3[1] + row3[2]*row3[2];
667 
668  ScalarT a_row2[3];
669 
670  const bool a0lea1 = a0 <= a1;
671 
672  a_row2[0] = if_then_else(a0lea1, row3[0], row2[0]);
673  a_row2[1] = if_then_else(a0lea1, row3[1], row2[1]);
674  a_row2[2] = if_then_else(a0lea1, row3[2], row2[2]);
675  const ScalarT ai_ai = ScalarT(1.0) / if_then_else(a0lea1, a1, a0);
676 
677  evec2[K_X] = k_row1[1]*a_row2[2] - k_row1[2]*a_row2[1];
678  evec2[K_Y] = k_row1[2]*a_row2[0] - k_row1[0]*a_row2[2];
679  evec2[K_Z] = k_row1[0]*a_row2[1] - k_row1[1]*a_row2[0];
680 
681  const ScalarT k_atr11 = cxx*k_row1[0] + cxy*k_row1[1] + czx*k_row1[2];
682  const ScalarT k_atr21 = cxy*k_row1[0] + cyy*k_row1[1] + cyz*k_row1[2];
683  const ScalarT k_atr31 = czx*k_row1[0] + cyz*k_row1[1] + czz*k_row1[2];
684 
685  const ScalarT a_atr12 = cxx*a_row2[0] + cxy*a_row2[1] + czx*a_row2[2];
686  const ScalarT a_atr22 = cxy*a_row2[0] + cyy*a_row2[1] + cyz*a_row2[2];
687  const ScalarT a_atr32 = czx*a_row2[0] + cyz*a_row2[1] + czz*a_row2[2];
688 
689  ScalarT rm2xx = (k_row1[0]*k_atr11 + k_row1[1]*k_atr21 + k_row1[2]*k_atr31)*ki_ki;
690  const ScalarT k_a_rm2xy = (k_row1[0]*a_atr12 + k_row1[1]*a_atr22 + k_row1[2]*a_atr32);
691  ScalarT rm2yy = (a_row2[0]*a_atr12 + a_row2[1]*a_atr22 + a_row2[2]*a_atr32)*ai_ai;
692  const ScalarT rm2xy_rm2xy = k_a_rm2xy*k_a_rm2xy*ai_ai*ki_ki;
693 
694  //
695  // Wilkinson shift
696  //
697  const ScalarT b = ScalarT(0.5)*(rm2xx-rm2yy);
698  const ScalarT sqrtTerm = MultiplySign( std::sqrt(b*b+rm2xy_rm2xy) ,b );
699  eval0 = rm2yy + b - sqrtTerm;
700 
701  eval1 = rm2xx + rm2yy - eval0;
702 
703  rm2xx -= eval0;
704  rm2yy -= eval0;
705 
706  const ScalarT rm2xx2 = rm2xx*rm2xx;
707  const ScalarT rm2yy2 = rm2yy*rm2yy;
708 
709  const ScalarT fac1 = if_then_else(rm2xx2 < rm2yy2, k_a_rm2xy*ai_ai, rm2xx);
710  const ScalarT fac2 = if_then_else(rm2xx2 < rm2yy2, rm2yy, ki_ki*k_a_rm2xy);
711 
712  evec0[0] = fac1*a_row2[0] - fac2*k_row1[0];
713  evec0[1] = fac1*a_row2[1] - fac2*k_row1[1];
714  evec0[2] = fac1*a_row2[2] - fac2*k_row1[2];
715 
716  const bool rm2xx2iszero = rm2xx2 == ScalarT(0.0);
717  const bool rm2xy_rm2xyiszero = rm2xy_rm2xy == ScalarT(0.0);
718  const bool both_zero = rm2xx2iszero && rm2xy_rm2xyiszero;
719 
720  // check degeneracy
721  evec0[0] = if_then_else(both_zero, a_row2[0], evec0[0]);
722  evec0[1] = if_then_else(both_zero, a_row2[1], evec0[1]);
723  evec0[2] = if_then_else(both_zero, a_row2[2], evec0[2]);
724 
725  evec1[K_X] = evec2[K_Y]*evec0[K_Z] - evec2[K_Z]*evec0[K_Y];
726  evec1[K_Y] = evec2[K_Z]*evec0[K_X] - evec2[K_X]*evec0[K_Z];
727  evec1[K_Z] = evec2[K_X]*evec0[K_Y] - evec2[K_Y]*evec0[K_X];
728 
729  eval0 += c1;
730  eval1 += c1;
731  eval2 += c1;
732 
733  const ScalarT c2tol = (c1*c1)*(-1.0e-30); // DJL where does the magic number come from?
734 
735  const bool c2lsmall_neg = c2 < c2tol;
736 
737  eval0 = if_then_else(c2lsmall_neg,eval0,c1);
738  eval1 = if_then_else(c2lsmall_neg,eval1,c1);
739  eval2 = if_then_else(c2lsmall_neg,eval2,c1);
740 
741  evec0[0] = if_then_else(c2lsmall_neg,evec0[0],ScalarT(1.0));
742  evec0[1] = if_then_else(c2lsmall_neg,evec0[1],ScalarT(0.0));
743  evec0[2] = if_then_else(c2lsmall_neg,evec0[2],ScalarT(0.0));
744 
745  evec1[0] = if_then_else(c2lsmall_neg,evec1[0],ScalarT(0.0));
746  evec1[1] = if_then_else(c2lsmall_neg,evec1[1],ScalarT(1.0));
747  evec1[2] = if_then_else(c2lsmall_neg,evec1[2],ScalarT(0.0));
748 
749  evec2[0] = if_then_else(c2lsmall_neg,evec2[0],ScalarT(0.0));
750  evec2[1] = if_then_else(c2lsmall_neg,evec2[1],ScalarT(0.0));
751  evec2[2] = if_then_else(c2lsmall_neg,evec2[2],ScalarT(1.0));
752 
753  return;
754 }
