154  std::vector<std::vector<int>> gather(const std::vector<int>& source, int root) const
155  {
156  if (this->get_rank() == root)
157  {
158  // Step 1: figure out how much data is going to be incoming from each rank
159  std::vector<int> recv_counts(this->get_size());
160  int sendcount = source.size();
161  // TO-DO: review documentation at
162  // https://www.mpich.org/static/docs/v3.1/www3/MPI_Gather.html
163  // https://www.mpich.org/static/docs/v3.1/www3/MPI_Gatherv.html
164  // to ensure that the function call is being passed the correct aruments.
165  MPI_Gather(&sendcount, 1, MPI_INT, recv_counts.data(), 1, MPI_INT, root, this->comm);
166 
167  // Step 2: Get data from all ranks as contiguous block of data
168  // destination.size() is the sum of the lengths of all incoming arrays.
169  std::vector<int> destination(std::accumulate(recv_counts.begin(), recv_counts.end(), 0));
170  std::vector<int> displacements(this->get_size(), 0);
171  std::partial_sum(recv_counts.begin(), recv_counts.end() - 1, displacements.data() + 1);
172  MPI_Gatherv(source.data(),
173  sendcount,
174  MPI_INT,
175  destination.data(),
176  recv_counts.data(),
177  displacements.data(),
178  MPI_INT,
179  root,
180  this->get_comm());
181 
182  // Step 3: Copy data from a contiguous block into a vector of vectors.
183  std::vector<std::vector<int>> result(this->get_size());
184  int *start_iter = destination.data(), *end_iter = destination.data() + recv_counts[0];
185  for (size_t i = 0; i < result.size(); ++i)
186  {
187  result[i] = std::vector<int>(start_iter, end_iter);
188  start_iter = end_iter;
189  end_iter += recv_counts[i];
190  }
191  // Return result
192  return result;
193  }
194  else
195  {
196  // Step 1 (see above)
197  int sendcount = source.size();
198  MPI_Gather(&sendcount, 1, MPI_INT, 0, 0, MPI_INT, root, this->get_comm());
199 
200  // Step 2 (see above)
201  MPI_Gatherv(source.data(), sendcount, MPI_INT, 0, 0, 0, MPI_INT, root, this->get_comm());
202 
203  // Return empty vector, as only the root rank will recieve incoming data.
204  return {};
205  }
206  }
207  std::vector<int> scatter(int root, const std::vector<std::vector<int>>& source) const
208  {
209  if (this->get_rank() == root)
210  {
211  // Step 1: notify each rank of how much data they should expect to recieve
212  int recvcount;
213  std::vector<int> send_counts(this->get_size());
214  for (size_t i = 0; i < send_counts.size(); ++i)
215  {
216  send_counts[i] = source[i].size();
217  }
218  MPI_Scatter(send_counts.data(), 1, MPI_INT, &recvcount, 1, MPI_INT, root, this->get_comm());
219 
220  // Step 2: send out data to each of the ranks.
221  std::vector<int> destination(recvcount, 0);
222  std::vector<int> displacements(this->get_size(), 0);
223  std::vector<int> source_made_contiguous(
224  std::accumulate(send_counts.begin(), send_counts.end(), 0), 0);
225 
226  std::partial_sum(send_counts.begin(), send_counts.end() - 1, displacements.data() + 1);
227  MPI_Scatterv(source.data(),
228  send_counts.data(),
229  displacements.data(),
230  MPI_INT,
231  destination.data(),
232  recvcount,
233  MPI_INT,
234  root,
235  this->get_comm());
236  }
237  else
238  {
239  int recvcount;
240  MPI_Scatter(0, 0, MPI_INT, &recvcount, 1, MPI_INT, root, this->get_comm());
241  std::vector<int> recvbuff(recvcount);
242  MPI_Scatterv(0, 0, 0, MPI_INT, recvbuff.data(), recvcount, MPI_INT, root, this->get_comm());
243  return recvbuff;
244  }
245  return std::vector<int>();

429  std::string catenate_gatherv_format(const std::string& str,
430  const std::string& _start,
431  const std::string& separator,
432  const std::string& _end) const
433  {
434  if (is_root())
435  {
436  std::vector<int> counts{gather_recieve(str.size())};
437  std::vector<int> displacements;
438  displacements.reserve(counts.size());
439  int displacement_accumulator = _start.size();
440  int separator_size = separator.size();
441  for (int count : counts)
442  {
443  displacements.emplace_back(displacement_accumulator);
444  displacement_accumulator += separator_size + count;
445  }
446  std::string dest((size_t)(displacement_accumulator + _end.size() - separator_size), ' ');
447  MPI_Gatherv(str.data(), /* data being sent */
448  str.size(), /* amount of data being sent */
449  MPI_CHAR, /* type of data being sent */
450  &dest[0], /* destination of data being sent */
451  counts.data(), /* expected amount of data to be recieved from each rank */
452  displacements.data(), /* displacements of revieced data */
453  MPI_CHAR, /* type of data being recieved */
454  this->get_root(), /* root */
455  this->get_comm() /* MPI Comm to use */
456  );
457  std::copy(_start.begin(), _start.end(), &dest[0]);
458  for (int i = 0, max = counts.size() - 1; i < max; ++i)
459  {
460  int offset = displacements[i] + counts[i];
461  std::copy(separator.begin(), separator.end(), &dest[offset]);
462  }
463  std::copy(_end.begin(), _end.end(), &dest[displacements.back() + counts.back()]);
464  return dest;
465  }
466  else
467  {
468  gather_send(str.size());
469  MPI_Gatherv(str.data(), /* data being sent */
470  str.size(), /* amount of data being sent */
471  MPI_CHAR, /* type of data being sent */
472  nullptr, /* destination of data being sent */
473  nullptr, /* expected amount of data to be recieved from each rank */
