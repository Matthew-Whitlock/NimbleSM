88  template <typename ViewT>
89  void ApplyInitialConditions(const ViewT reference_coordinates,
90  ViewT velocity) {
91 
92  for (unsigned int i_bc=0 ; i_bc<boundary_conditions_.size() ; i_bc++) {
93 
94  BoundaryCondition & bc = boundary_conditions_.at(i_bc);
95 
96  if (bc.bc_type_ == BoundaryCondition::INITIAL_VELOCITY) {
97  int node_set_id = bc.node_set_id_;
98  int coordinate = bc.coordinate_;
99  bool has_expression = bc.has_expression_;
100  std::vector<int> const & node_set = node_sets_[node_set_id];
101  if (!has_expression) {
102  double magnitude = bc.magnitude_;
103  for (unsigned int n=0 ; n<node_set.size() ; n++) {
104  velocity(node_set[n],coordinate) = magnitude;
105  }
106  }
107  else {
108  for (unsigned int n=0 ; n<node_set.size() ; n++) {
109  bc.expression_.x = reference_coordinates(node_set[n],0);
110  bc.expression_.y = reference_coordinates(node_set[n],1);
111  bc.expression_.z = 0.0;
112  if (dim_ == 3) {
113  bc.expression_.z = reference_coordinates(node_set[n],2);
114  }
115  bc.expression_.t = 0.0;
116  velocity(node_set[n],coordinate) = bc.expression_.eval();
117  }
118  }

123  template <typename ViewT>
124  void ApplyKinematicBC(double time_current,
125  double time_previous,
126  const ViewT reference_coordinates,
127  ViewT displacement,
128  ViewT velocity) {
129 
130  double delta_t = time_current - time_previous;
131 
132  for (unsigned int i_bc=0 ; i_bc<boundary_conditions_.size() ; i_bc++) {
133 
134  BoundaryCondition & bc = boundary_conditions_[i_bc];
135  int node_set_id = bc.node_set_id_;
136  int coordinate = bc.coordinate_;
137  bool has_expression = bc.has_expression_;
138  std::vector<int> const & node_set = node_sets_[node_set_id];
139 
140  if (bc.bc_type_ == BoundaryCondition::PRESCRIBED_VELOCITY) {
141  if (!has_expression) {
142  double velocity_magnitude = bc.magnitude_;
143  for (unsigned int n=0 ; n<node_set.size() ; n++) {
144  velocity(node_set[n],coordinate) = velocity_magnitude;
145  if (time_integration_scheme_ == QUASISTATIC) {
146  displacement(node_set[n],coordinate) += velocity_magnitude * delta_t;
147  }
148  }
149  }
150  else {
151  for (unsigned int n=0 ; n<node_set.size() ; n++) {
152  bc.expression_.x = reference_coordinates(node_set[n],0);
153  bc.expression_.y = reference_coordinates(node_set[n],1);
154  bc.expression_.z = 0.0;
155  if (dim_ == 3) {
156  bc.expression_.z = reference_coordinates(node_set[n],2);
157  }
158  bc.expression_.t = time_current;
159  double velocity_magnitude = bc.expression_.eval();
160  velocity(node_set[n],coordinate) = velocity_magnitude;
161  if (time_integration_scheme_ == QUASISTATIC) {
162  displacement(node_set[n],coordinate) += velocity_magnitude * delta_t;
163  }
164  }
165  }
166  }
167 
168  else if (bc.bc_type_ == BoundaryCondition::PRESCRIBED_DISPLACEMENT && delta_t > 0.0) {
169  if (!has_expression) {
170  double displacement_magnitude = bc.magnitude_;
171  for (unsigned int n=0 ; n<node_set.size() ; n++) {
172  velocity(node_set[n],coordinate) = (displacement_magnitude - displacement(node_set[n],coordinate))/delta_t;
173  if (time_integration_scheme_ == QUASISTATIC) {
174  displacement(node_set[n],coordinate) = displacement_magnitude;
175  }
176  }
177  }
178  else {
179  for (unsigned int n=0 ; n<node_set.size() ; n++) {
180  bc.expression_.x = reference_coordinates(node_set[n],0);
181  bc.expression_.y = reference_coordinates(node_set[n],1);
182  bc.expression_.z = 0.0;
183  if (dim_ == 3) {
184  bc.expression_.z = reference_coordinates(node_set[n],2);
185  }
186  bc.expression_.t = time_current;
187  double displacement_magnitude = bc.expression_.eval();
188  velocity(node_set[n],coordinate) = (displacement_magnitude - displacement(node_set[n],coordinate))/delta_t;
189  if (time_integration_scheme_ == QUASISTATIC) {
190  displacement(node_set[n],coordinate) = displacement_magnitude;
191  }
192  }
193  }
194  }
195  }
