60  inline
61  double TriangleArea(const double * const pt_1,
62  const double * const pt_2,
63  const double * const pt_3) {
64 
65  double a[3], b[3], cross[3], area;
66  for (int i=0 ; i<3 ; ++i) {
67  a[i] = pt_2[i] - pt_1[i];
68  b[i] = pt_3[i] - pt_1[i];
69  }
70  cross[0] = b[1]*a[2] - b[2]*a[1];
71  cross[1] = b[0]*a[2] - b[2]*a[0];
72  cross[2] = b[0]*a[1] - b[1]*a[0];
73  area = 0.5 * std::sqrt(cross[0]*cross[0] + cross[1]*cross[1] + cross[2]*cross[2]);
74  return area;
75  }

126  void SetCoordinates(const double * const coord) {
127  for (unsigned int i_node=0 ; i_node<num_nodes_ ; i_node++){
128  for (int i_dof = 0 ; i_dof < 3 ; i_dof++) {
129  coord_[3*i_node + i_dof] = 0.0;
130  }
131  for (auto const & entry : map_to_mesh_nodes_.at(i_node)) {

133  double multiplier = entry.second;
134  for (int i_dof = 0 ; i_dof < 3 ; i_dof++) {
135  coord_[3*i_node + i_dof] += multiplier * coord[3*node_id + i_dof];
136  }

177  void SetBoundingBox() {
178 
179  centroid_[0] = centroid_[1] = centroid_[2] = 0.0;
180 
181  bounding_box_x_min_ = std::numeric_limits<double>::max();
182  bounding_box_x_max_ = std::numeric_limits<double>::lowest();
183  bounding_box_y_min_ = std::numeric_limits<double>::max();
184  bounding_box_y_max_ = std::numeric_limits<double>::lowest();
185  bounding_box_z_min_ = std::numeric_limits<double>::max();
186  bounding_box_z_max_ = std::numeric_limits<double>::lowest();
187 
188  for (int i_node=0 ; i_node<num_nodes_ ; i_node++) {
189 
190  centroid_[0] += coord_[3*i_node];
191  centroid_[1] += coord_[3*i_node+1];
192  centroid_[2] += coord_[3*i_node+2];
193 
194  if (coord_[3*i_node] < bounding_box_x_min_)
195  bounding_box_x_min_ = coord_[3*i_node];
196  if (coord_[3*i_node] > bounding_box_x_max_)
197  bounding_box_x_max_ = coord_[3*i_node];
198  if (coord_[3*i_node+1] < bounding_box_y_min_)
199  bounding_box_y_min_ = coord_[3*i_node+1];
200  if (coord_[3*i_node+1] > bounding_box_y_max_)
201  bounding_box_y_max_ = coord_[3*i_node+1];
202  if (coord_[3*i_node+2] < bounding_box_z_min_)
203  bounding_box_z_min_ = coord_[3*i_node+2];
204  if (coord_[3*i_node+2] > bounding_box_z_max_)
205  bounding_box_z_max_ = coord_[3*i_node+2];
206  }
207 
208  centroid_[0] /= num_nodes_;
209  centroid_[1] /= num_nodes_;
210  centroid_[2] /= num_nodes_;
211 
212  double inflation_length = 0.15 * char_len_;
213 
214  bounding_box_x_min_ -= inflation_length;
215  bounding_box_x_max_ += inflation_length;
216  bounding_box_y_min_ -= inflation_length;
217  bounding_box_y_max_ += inflation_length;
218  bounding_box_z_min_ -= inflation_length;
219  bounding_box_z_max_ += inflation_length;
220  }

222  void ComputeNodalContactForces(const double * const contact_force,
223  const double * const closest_point_projection) {
224 
225  std::vector<double> N(num_nodes_);
226 
227  if (entity_type_ == NODE) {
228  N[0] = 1.0;
229  }
230  else if (entity_type_ == TRIANGLE) {
231  // Find the natural coordinates of the closest_point_projection
232  double area_1 = TriangleArea(closest_point_projection, &coord_[3], &coord_[6]);
233  double area_2 = TriangleArea(&coord_[0], closest_point_projection, &coord_[6]);
234  double full_area = TriangleArea(&coord_[0], &coord_[3], &coord_[6]);
235  N[0] = area_1/full_area;
236  N[1] = area_2/full_area;
237  N[2] = 1.0 - N[0] - N[1];
238  }
239 
240  for (int i_node = 0; i_node < num_nodes_ ; ++i_node) {
241  for (int i=0 ; i<3 ; ++i) {
242  force_[3*i_node + i] = N[i_node] * contact_force[i];
243  }
244  }
245  }
246

270  std::vector< std::vector< std::pair<int, double> > > map_to_mesh_nodes_;
