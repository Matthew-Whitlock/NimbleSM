85 std::vector<int> PackIDSpace(const std::vector<int>& raw_node_ids,
86  int max_nodes_assigned_to_a_rank,
87  const mpicontext& context)
88 {
89  if (context.is_root())
90  {
91  std::map<int, int> lookup;
92  std::vector<int> buffer(max_nodes_assigned_to_a_rank);
93  int sane_node_id = 0;
94  for (int node : raw_node_ids)
95  if (lookup.count(node) == 0)
96  {
97  lookup[node] = sane_node_id;
98  ++sane_node_id;
99  }
100 
101  for (int rank = 1, context_size = context.get_size(); rank < context_size; ++rank)
102  {
103  MPI_Status recv_status;
104  int recv_size;
105  int err = MPI_Recv(
106  buffer.data(), buffer.size(), MPI_INT, rank, 0xfae, context.get_comm(), &recv_status);
107  if (err != MPI_SUCCESS)
108  throw std::logic_error("Bad recv in PackIDSpace");
109  MPI_Get_count(&recv_status, MPI_INT, &recv_size);
110  for (int i = 0; i < recv_size; ++i)
111  {
112  int node = buffer[i];
113  if (lookup.count(node) == 0)
114  {
115  lookup[node] = sane_node_id;
116  ++sane_node_id;
117  }
118  buffer[i] = lookup[node];
119  }
120  MPI_Send(buffer.data(), recv_size, MPI_INT, rank, 0xfae, context.get_comm());
121  }
122  buffer.resize(raw_node_ids.size());
123  buffer.shrink_to_fit();
124  for (int i = 0, max = buffer.size(); i < max; ++i)
125  {
126  buffer[i] = lookup[raw_node_ids[i]];
127  }
128  // std::cout << "rank " << context.get_rank() << " new_ids: " << buffer << std::endl;
129  return buffer;
130  }
131  else
132  {
133  std::vector<int> new_ids(raw_node_ids.size());
134  MPI_Send(raw_node_ids.data(), raw_node_ids.size(), MPI_INT, 0, 0xfae, context.get_comm());
135  MPI_Recv(
136  new_ids.data(), new_ids.size(), MPI_INT, 0, 0xfae, context.get_comm(), MPI_STATUS_IGNORE);
137  // std::cout << "rank " << context.get_rank() << " new_ids: " << new_ids << std::endl;
138  return new_ids;
139  }

141 int GetMaximumNodeId(const std::vector<int>& global_node_ids, const mpicontext& context)

164 template<class Key, class Val>
165 std::vector<std::pair<Key, std::vector<Val>>> GroupConsecutive(
166  const std::vector<std::pair<Key, Val>>& lst)
167 {
168  if (lst.size() == 0)
169  return {};
170  std::vector<int> key_ids;
171  key_ids.reserve(lst.size() + 1);
172  const Key* group_key = &lst[0].first;
173  int max_key_id = 0;
174  int lst_size = lst.size();
175  for (const std::pair<Key, Val>& p : lst)
176  {
177  if (p.first != *group_key)
178  {
179  ++max_key_id;
180  group_key = &p.first;
181  }
182  key_ids.push_back(max_key_id);
183  }
184  key_ids.push_back(max_key_id + 1);
185  std::vector<std::pair<Key, std::vector<Val>>> groups;
186  groups.reserve(max_key_id + 1);
187  {
188  int prev_key_id = 0;
189  int group_size = 1;
190  for (int i = 1; i <= lst_size; ++i)
191  {
192  if (key_ids[i] != prev_key_id)
193  {
194  groups.emplace_back(lst[i - 1].first, std::vector<Val>(group_size));
195  auto& new_elem = groups.back();
196  const std::pair<Key, Val>* src_ptr = &lst[i - group_size];
197  for (Val& q : new_elem.second)
198  {
199  q = src_ptr->second;
200  ++src_ptr;
201  }
202  group_size = 1;
203  prev_key_id = key_ids[i];
204  }
205  else
206  {
207  ++group_size;
208  }
209  }
210  }
211  return groups;
212 }
