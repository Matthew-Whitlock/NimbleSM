144 std::vector<arrayview_t<T>> partition_into_arrayviews(std::vector<T>& values, const list_t& counts)
145 {
146  std::vector<arrayview_t<T>> views;
147  views.reserve(counts.size());
148  size_t displacement = 0;
149  for (auto& count : counts)
150  {
151  views.emplace_back(values, displacement, count);
152  displacement += count;
153  }
154  return views;
155 }
